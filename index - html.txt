    <script>
        // Modal functions
        function openModal(modalId) {
            document.getElementById(modalId).style.display = "block";
            document.body.style.overflow = "hidden";
        }
        
        function closeModal(modalId) {
            document.getElementById(modalId).style.display = "none";
            document.body.style.overflow = "auto";
        }
        
        // Close modal when clicking outside
        window.onclick = function(event) {
            if (event.target.className === "modal") {
                event.target.style.display = "none";
                document.body.style.overflow = "auto";
            }
        }
        
        // Mobile menu toggle
        function toggleMobileMenu() {
            const menu = document.getElementById("mobileMenu");
            if (menu.classList.contains("hidden")) {
                menu.classList.remove("hidden");
            } else {
                menu.classList.add("hidden");
            }
        }

        // Document management functions
        let documents = [
            { id: 1, name: "Modelo de Contrato", type: "pdf", category: "familyArticle" },
            { id: 2, name: "Formulário de Procuração", type: "docx", category: "laborArticle" },
            { id: 3, name: "Checklist Documentos", type: "xlsx", category: "criminalArticle" }
        ];

        function renderFiles() {
            const fileItems = document.getElementById('fileItems');
            fileItems.innerHTML = '';
            
            documents.forEach(doc => {
                const fileItem = document.createElement('div');
                fileItem.className = 'file-item';
                fileItem.innerHTML = `
                    <div>
                        <span class="font-medium">${doc.name}</span>
                        <span class="text-gray-500 text-sm ml-2">.${doc.type}</span>
                        <span class="text-xs bg-gray-200 px-2 py-1 rounded ml-2">${getCategoryName(doc.category)}</span>
                    </div>
                    <div class="file-actions">
                        <button onclick="editFileName(${doc.id})" class="text-blue-600 hover:text-blue-800">
                            <i class="fas fa-edit"></i>
                        </button>
                        <button onclick="deleteFile(${doc.id})" class="text-red-600 hover:text-red-800">
                            <i class="fas fa-trash"></i>
                        </button>
                    </div>
                `;
                fileItems.appendChild(fileItem);
            });
            
            // Update article contents with related documents
            updateArticleContents();
        }

        function getCategoryName(categoryId) {
            switch(categoryId) {
                case 'familyArticle': return 'Família';
                case 'laborArticle': return 'Trabalhista';
                case 'criminalArticle': return 'Criminal';
                default: return 'Geral';
            }
        }

        function updateArticleContents() {
            // Clear all article contents first
            document.querySelectorAll('.article-content').forEach(el => {
                el.innerHTML = '';
            });
            
            // Group documents by category
            const documentsByCategory = {};
            documents.forEach(doc => {
                if (!documentsByCategory[doc.category]) {
                    documentsByCategory[doc.category] = [];
                }
                documentsByCategory[doc.category].push(doc);
            });
            
            // Add documents to their respective articles
            for (const [categoryId, docs] of Object.entries(documentsByCategory)) {
                const articleContent = document.getElementById(categoryId);
                if (articleContent) {
                    const docsList = document.createElement('div');
                    docsList.className = 'space-y-3';
                    
                    docs.forEach(doc => {
                        const docItem = document.createElement('div');
                        docItem.className = 'border-b pb-2';
                        docItem.innerHTML = `
                            <h4 class="font-medium">${doc.name}</h4>
                            <div class="flex justify-between items-center mt-1">
                                <span class="text-sm text-gray-500">.${doc.type}</span>
                                <button onclick="downloadDocument(${doc.id})" class="text-blue-600 hover:text-blue-800 text-sm">
                                    <i class="fas fa-download mr-1"></i>Baixar
                                </button>
                            </div>
                        `;
                        docsList.appendChild(docItem);
                    });
                    
                    articleContent.appendChild(docsList);
                }
            }
        }

        function uploadFile() {
            const fileInput = document.getElementById('fileInput');
            const fileNameInput = document.getElementById('fileName');
            const articleCategory = document.getElementById('articleCategory');
            
            if (fileNameInput.value.trim() === '') {
                alert('Por favor, insira um nome para o documento');
                return;
            }
            
            if (fileInput.files.length === 0) {
                alert('Por favor, selecione um arquivo');
                return;
            }
            
            if (!articleCategory.value) {
                alert('Por favor, selecione o artigo relacionado');
                return;
            }
            
            const file = fileInput.files[0];
            const fileType = file.name.split('.').pop().toLowerCase();
            
            // In a real application, you would upload the file to a server here
            // For this demo, we'll just add it to our documents array
            const newDoc = {
                id: documents.length + 1,
                name: fileNameInput.value.trim(),
                type: fileType,
                category: articleCategory.value
            };
            
            documents.push(newDoc);
            renderFiles();
            
            // Reset form
            fileNameInput.value = '';
            fileInput.value = '';
            articleCategory.value = '';
            
            alert('Documento enviado com sucesso!');
        }

        function editFileName(id) {
            const doc = documents.find(d => d.id === id);
            if (!doc) return;
            
            const newName = prompt('Editar nome do documento:', doc.name);
            if (newName && newName.trim() !== '') {
                doc.name = newName.trim();
                renderFiles();
            }
        }

        function deleteFile(id) {
            if (confirm('Tem certeza que deseja excluir este documento?')) {
                documents = documents.filter(d => d.id !== id);
                renderFiles();
            }
        }

        function downloadDocument(id) {
            const doc = documents.find(d => d.id === id);
            if (!doc) return;
            
            alert(`Simulando download do documento: ${doc.name}.${doc.type}`);
            // In a real application, this would trigger the file download
        }

        function toggleArticle(articleId) {
            event.preventDefault();
            const articleContent = document.getElementById(articleId);
            if (articleContent.style.display === 'block') {
                articleContent.style.display = 'none';
            } else {
                articleContent.style.display = 'block';
            }
        }

        // Initialize file list when page loads
        document.addEventListener('DOMContentLoaded', renderFiles);
    </script>
